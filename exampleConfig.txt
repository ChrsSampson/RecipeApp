This is what the config.js file should look like
create a config.js file and fill in the values for your enviroment

const config = {
  port: [your port],
  db:{
    host: 'localhost',
    user: 'recipe',
    password: '[your database password]'
  },
  session: {
    secret: '[your secret]',
  }
}

module.exports = config;
